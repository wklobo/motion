CREATE TABLE `T_Events` (
  `KeyID` int(11) NOT NULL AUTO_INCREMENT COMMENT 'Primärindex',
  `Saved` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT 'Zeitpunkt der Speicherung',
  `Event` text COLLATE utf8_general_mysql500_ci NOT NULL COMMENT 'Verzeichnisname des Ereignisses',
  `Datum` date NOT NULL COMMENT 'Datum des Ereignisses',
  `Zeit` time NOT NULL COMMENT 'Beendigung des Ereignisses',
  `Size` int(10) unsigned NOT NULL COMMENT 'Summe des belegten Speichers im Verzeichnis',
  `Bemerkung` tinyblob COMMENT 'einfach nur Bemerkungen',
  PRIMARY KEY (`KeyID`),
  KEY `idxEvent` (`Event`(25)),
  KEY `idxDatum` (`Datum`),
  KEY `idxZeit` (`Datum`,`Zeit`),
  KEY `idxSize` (`Size`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_general_mysql500_ci COMMENT='Ereignistabelle für Motion'
